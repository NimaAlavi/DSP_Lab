define myConv function:
function convOut = myConv(h, x)
size_x = length(x);
h_pad = [h zeros(1, size_x - 1)];
c = [h(1) zeros(1, size_x - 1)];
H = toeplitz(c, h_pad);
convOut = x * H;
end
